<TabItem value="php" label="PHP (Joomla Framework)">

```php title="Get List of Banners"
  use Joomla\Http\HttpFactory;
  use Joomla\Uri\Uri;

  $http = (new HttpFactory())->getAvailableDriver();
  $url  = 'http://www.mysite.com/api/v1';
  $uri  = new Uri($url);

  $token = {api_key};

  $headers = [
    'Accept: application/vnd.api+json',
    'Content-Type: application/json',
    sprintf('X-Joomla-Token: %s', trim($token)),
  ];

  $timeout = 30;

  $uri->setPath('banners');

  $response = $http->request('GET', $uri, null, $headers, $timeout);

  echo $response->body;
```

</TabItem>
